    def reversed_list_index(lst, target, index=-1):
    if index < -len(lst):
        return -1
    if lst[index] == target:
        return len(lst) + index
    return reversed_list_index(lst, target, index - 1)

    def dup_count(string):

    value_count = {}
    
    duplicates = []

    # Count occurrences of each value
    
    
    for value in string:
        
        if value in value_count:
            
            value_count[value] += 1
            
        else:
            
            value_count[value] = 1

    # Find duplicates and their occurrences
    
    
    for value, count in value_count.items():
        
        if count > 1:
            
            duplicates.append((value, count))

    return duplicates
